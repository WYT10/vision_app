cmake_minimum_required(VERSION 3.16)
project(camera_combo_app LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(OpenCV REQUIRED COMPONENTS core highgui imgproc imgcodecs videoio aruco)
find_package(nlohmann_json REQUIRED)

add_executable(camera_combo_app main.cpp)
target_link_libraries(camera_combo_app PRIVATE ${OpenCV_LIBS} nlohmann_json::nlohmann_json)
target_include_directories(camera_combo_app PRIVATE ${OpenCV_INCLUDE_DIRS})
